% References that should open in a browser or in a new PDF viewer
% are defined here. This includes external URLs or links to PDFs
% that are distributed with this text.
%
% Citations for journals, conferences, etc., that are not distributed
% with this text should be defined statically in References.bib.
%
% This TeX file creates a BiBTeX entry for a reference. Each BiBTeX
% BiBTeX entry is associated to a URL or local PDF document. The
% BiBTeX file is generated when this file is processed during TeX
% build, so if you change this document you should first build,
% then run BiBTeX (which inputs the newly generated .bib file), and
% lastly run build again.
%
% Concisely: IF YOU CHANGE THIS DOCUMENT, then
%            Build - BiBTeX - Build
%
% Use Case: Create a new BiBTeX entry and PDF link
%   \newdocref[bibliography label][author][title][month][year][howpublished][url]
% Requires:
%   Corresponding PDF file \docpath/[bibliography label].pdf (colons ':' replaced by underscores '_')
% Errors:
%   If \docpath/[bibliography label].pdf (colons ':' replaced by underscores '_') does not exist,
%   an error will be thrown.
%
% Use Case: Create a new BiBTeX entry and URL link
%   \newlinkref[bibliography label][author][title][month][year][howpublished][url]
%
% Troubleshooting:
%   + Any superfluous characters in this file will be printed on page 1 of the document
%   + Any use of '=' character in a URL will cause truncation;
%       use \urlEqual instead and use double brackets {{http://...}} around the URL

% Workaround for having an equal sign in a URL (using '=' will truncate due to pdfkeys parsing)
\newcommand{\urlEqual}{=}
